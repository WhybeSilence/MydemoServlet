    subgraph UserAuth [1. 用户认证与访问控制]
        direction TB
        R_Start(打开注册页) --> R_Input[输入用户名密码]
        R_Input --> R_Valid{用户名合法?}
        R_Valid -- 否 --> R_Err[提示错误]
        R_Valid -- 是 --> R_Post[POST /register]
        R_Post --> R_Check{用户名存在?}
        R_Check -- 存在 --> R_409[返回 409]
        R_Check -- 不存在 --> R_Save[BCrypt加密并入库]
        R_Save --> R_200[注册成功 跳转登录]

        L_Start(打开首页/登录页) --> L_Input[提交 /login POST]
        L_Input --> L_Query[查询 sys_user 表]
        L_Query --> L_Exist{用户是否存在?}
        L_Exist -- 否 --> L_401[返回 401]
        L_Exist -- 是 --> L_Pass{密码匹配?}
        L_Pass -- 否 --> L_401
        L_Pass -- 是 --> L_Session[创建 Session/Cookie]
        L_Session --> L_Role{角色判断}
        L_Role -- 管理员 --> L_Admin[跳转 /admin]
        L_Role -- 店主 --> L_Shop[跳转 /shop_owner]
        L_Role -- 用户 --> L_Home[跳转 home.html]

        F_Req(用户请求) --> F_Path{路径检查}
        F_Path -- 静态资源/白名单 --> F_Pass[直接放行]
        F_Path -- 其他路径 --> F_Sess{Session有效?}
        F_Sess -- 否 --> F_Ajax{Ajax请求?}
        F_Ajax -- 是 --> F_JSON[返回 401 JSON]
        F_Ajax -- 否 --> F_Login[重定向登录页]
        F_Sess -- 是 --> F_Pass
    end

    subgraph ShopBrowse [2. 店铺浏览与商品展示]
        direction TB
        B_Home(访问 home.html) --> B_List[GET /shopList]
        B_List --> B_DB[(SQL: shop JOIN sys_user)]
        B_DB --> B_Render[渲染店铺卡片]
        B_Render --> B_Click[点击店铺 跳转 shopDetail]
        B_Click --> B_Det[GET /shopDetail?shopId]
        B_Det --> B_CheckRole{是否店主本人?}
        B_CheckRole -- 是 --> B_All[显示全部商品 含待审核/拒绝]
        B_CheckRole -- 否 --> B_Audit[仅显示已审核 audit_status=1]
    end

    subgraph ProductAudit [3. 商品管理与审核机制]
        direction TB
        P_List(店主管理页) --> P_Add[填写商品信息]
        P_Add --> P_Post[POST /product action=add]
        P_Post --> P_DB1[入库 audit_status=0]
        P_DB1 --> P_Msg1[发送审核通知消息]
        
        A_Start(管理员审核页) --> A_Get[GET /admin/audit]
        A_Get --> A_Wait[展示待审核列表]
        A_Wait --> A_Op{审核操作}
        A_Op -- 批准 --> A_App[更新 status=1]
        A_Op -- 拒绝 --> A_Rej[更新 status=2 + 原因]
        A_App --> A_MsgP[写入成功消息]
        A_Rej --> A_MsgR[写入拒绝消息]
    end

    subgraph WishlistStock [4. 愿望单与库存变动]
        direction TB
        W_Add(点击加入愿望单) --> W_Login{已登录?}
        W_Login -- 否 --> W_401[返回 401]
        W_Login -- 是 --> W_Stock{库存 > 0?}
        W_Stock -- 否 --> W_Empty[返回 400 售罄]
        W_Stock -- 是 --> W_Exist{已在愿望单?}
        W_Exist -- 是 --> W_Dup[返回 400 重复]
        W_Exist -- 否 --> W_Insert[插入 wishlist 表]
        W_Insert --> W_Dec[商品库存 stock - 1]
        
        W_View(查看愿望单) --> W_VList[GET /wishlist]
        W_VList --> W_Del[移除商品]
        W_Del --> W_DB_D[删除记录]
        W_DB_D --> W_Inc[商品库存 stock + 1]
    end

    subgraph Notifications [5. 系统消息通知]
        direction TB
        N_Trigger(系统事件触发) --> N_Type{事件类型}
        N_Type -- 审核通过 --> N_P[店主消息: 已通过]
        N_Type -- 审核拒绝 --> N_R[店主消息: 已拒绝+原因]
        N_Type -- 提交审核 --> N_S[店主消息: 提交成功]
        N_P & N_R & N_S --> N_DB[(Insert system_message)]
        
        N_User(店主查看消息) --> N_Fetch[GET /messages]
        N_Fetch --> N_List[按时间倒序展示]
    end

    subgraph DataFlow [6. 数据流向总览]
        DF_UI[前端层: home/detail/admin] <--> DF_Srv[Servlet 控制器层]
        DF_Srv <--> DF_Fil[Filter 过滤层]
        DF_Fil <--> DF_Model[实体层: User/Shop/Prod/Wish/Msg]
        DF_Model <--> DF_DB[(MySQL: shopdemo)]
    end

    %% 模块间连线

    P_Msg1 & A_MsgP & A_MsgR -.-> N_Trigger
    W_Insert -.-> DF_DB
